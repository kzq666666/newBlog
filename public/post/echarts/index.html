// 柱状图
echarts.init(document.getElementById('bar')).setOption({
        title:{
          text:'柱状图'
        },
        tooltip:{},
        legend:{
          data:['销量']
        },
        xAxis:{
          data:["衬衫","羊毛衫","雪纺衫","裤子","高跟鞋","袜子"]
        },
        yAxis:{},
        series: [{
            type: 'bar',
            name: '销量',
            data: [5, 20, 36, 10, 10, 20]
        }]
});

// 饼状图

        title:{
          text:'饼状图'
        },
        series: {
            name: '访问来源',
            type: 'pie',
            radius:'60%',
            data: [
                {value:235, name:'视频广告'},
                {value:274, name:'联盟广告'},
                {value:310, name:'邮件营销'},
                {value:335, name:'直接访问'},
                {value:400, name:'搜索引擎'}
            ]
        }
});
